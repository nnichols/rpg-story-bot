001  (ns rpg-story-bot.forms
002    (:require [rpg-story-bot.story-elements.beings :as sb]
003              [rpg-story-bot.story-elements.titles :as st]
004              [rpg-story-bot.story-elements.player-attributes :as pa]
005              [rpg-story-bot.utility :as util])
006    (:gen-class))
007  
008  (defn since-running
009    "Generate a quest of the following form:
010     Since __ we have been running from __"
011    []
012    (str "Ever since the "
013         (rand-nth sb/humanoids)
014         " attacked, we have been running from "
015         (st/title-individual (rand-nth sb/humanoid-individuals))))
016  
017  (defn into-a-bar
018    "Generate a plot hook of the following form:
019    A __, ___, and ____ walk into a bar. Only one walks out"
020    []
021    (str "A "
022         (util/get-random-map-vector pa/races-map)
023         ", "
024         (util/get-random-map-vector pa/races-map)
025         ", and "
026         (util/get-random-map-vector pa/races-map)
027         " walk into a bar. Only one walks out."))
028  
029  (defn next-character
030    "Generate a random character"
031    []
032    (str "When my current character dies, I'm going to roll a "
033         (rand-nth pa/alignments)
034         " "
035         (util/get-random-map-vector pa/races-map)
036         " "
037         (util/get-random-map-vector pa/classes-map)))
038  
039  (defn random-message
040    "Generate a random tweet derived from the above forms"
041    []
042    (condp = (rand-int 3)
043      1 (since-running)
044      2 (into-a-bar)
045      (next-character)))
