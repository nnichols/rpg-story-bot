001  (ns rpg-story-bot.core
002    (:require [rpg-story-bot.forms :as fm]
003              [rpg-story-bot.twitter-interface :as twit])
004    (:gen-class))
005  
006  (defn build-messages
007    "Create n random messages"
008    [n]
009    (take n (repeatedly #(fm/random-message))))
010  
011  (defn tweetable-message
012    "Takes a collection of messages and returns the first one that is tweetable,
013      if no tweets match, return nil"
014    [tweet-list]
015    (loop [running-list tweet-list]
016      (if (twit/check-length (first running-list))
017        (first running-list)
018        (recur (rest running-list)))))
019  
020  (defn -main
021    "Create a message and tweet it out"
022    [& args]
023    (twit/tweet (tweetable-message (build-messages 10)))
024    (twit/close)
025    (System/exit 0))
