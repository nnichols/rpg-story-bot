061  (defn apply-prefix-title
062    "Given person, return that same person with a prefix title"
063    [person]
064    (str (rand-nth prefix-titles) " " person))
065  
066  (defn apply-suffix-title
067    "Given person, return that same person with a suffix title"
068    [person]
069    (str person " " (rand-nth suffix-titles)))
070  
071  (defn apply-both-titles
072    "Given person, return that same person with both a prefix
073     and a suffix title"
074    [person]
075    (apply-prefix-title (apply-suffix-title person)))
076  
077  (defn title-individual
078    "Given person, return that same person with either a prefix title,
079        a suffix titles or both title forms added"
080    [person]
081    (condp = (rand-int 3)
082      1 (apply-prefix-title person)
083      2 (apply-suffix-title person)
084      (apply-both-titles person)))
